\section{Formalae and Constants}
\subsection*{Laws of indices}
$$
\begin{array}{l}
a^m \times a^n=a^{(m+n)} \\
\frac{a^m}{a^n}=a^{(m-n)} \\
\left(a^n\right)^n=a^m
\end{array}
$$


\subsection*{Laws of logarithms}
$$
\begin{array}{l}
\log A B=\log A+\log B \\
\log \frac{A}{B}=\log A-\log B \\
\log A^n=n \log A
\end{array}
$$
